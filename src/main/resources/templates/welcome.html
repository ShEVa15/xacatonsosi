// PSB Platform API Secure Transaction
const initiateSecureTransaction = async (userId, amount) => {
  try {
    console.log(`Processing transaction for ${userId}...`);

    const response = await fetch('https://api.psb.ru/v2/finance', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${TOKEN}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({ account: userId, value: amount })
    });

    const data = await response.json();
    return { success: true, id: data.transactionId };

  } catch (error) {
    return { success: false, error: error.message };
  }
};
